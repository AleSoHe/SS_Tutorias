\documentclass[margin=5pt]{standalone}
\usepackage{tikz,pgfplots}

\begin{document}
    \def \minx {-1.2}
    \def \maxx {3.2}
    \def \miny {-2.2}
    \def \maxy {2.2}
    \def \delta {0.5}

    \begin{tikzpicture}[scale=1]
        % Draw axis
        \draw[step=1,gray,ultra thin] (\minx,\miny) grid (\maxx,\maxy);
        \draw [->] (\minx,0) -- (\maxx+\delta,0) node [right,scale=0.9] {\scriptsize $Re\{z\}$};
        \draw [->] (0,\miny) -- (0,\maxy+\delta) node [above,scale=0.9] {\scriptsize $Im\{z\}$};

        % Draw semicircle
        \begin{scope}
            \clip (1,-1) rectangle (2,1);
            \draw [fill=green,draw=none,opacity=0.5] (1,0) circle (1);
            \draw (1,0) circle (1);
        \end{scope}

        % Draw dashed semicircle
        \begin{scope}
            \clip (0,-1) rectangle (1,1);
            \draw [dashed] (1,0) circle (1);
        \end{scope}

        % Draw rectangle
        \draw [fill=green,draw=none,opacity=0.5] (0,-1) rectangle (1,1);

        % Draw rectangle lines
        \draw (0,-1) -- (1,-1);
        \draw (0,1) -- (1,1);

        % Axis numbers
        \node[below left] at (0,0) {\footnotesize $0$};
        \foreach \y in {-1,...,-1,1,...,1}
            \draw[shift={(0,\y)}, very thin] (2pt,0pt) -- (-2pt,0pt) node[below left] {\scriptsize $\y$};
        \foreach \x in {1,...,2}
            \draw[shift={(\x,0)}, very thin] (0pt,2pt) -- (0pt,-2pt) node[above right] {\scriptsize $\x$};


    \end{tikzpicture}
\end{document}
